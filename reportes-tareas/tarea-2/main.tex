        {\textsc{Ecuaciones Diferenciales Parciales II} \\
        \medskip 
        \large{Tarea 2}}
    \end{center}
    
    \bigskip
    
    \begin{enumerate}
        \item Diseñar un algoritmo en Python que resuelva ecuaciones
        diferenciales con el método de Descomposición de Adomian en general.

        \sol Se implemento un módulo en Python para el manejo de ecuaciones
        diferenciales y sistemas de ecuaciones diferenciales, este es el módulo
        \textit{eqsolver} dentro del repositorio de esta materia creado en la
        Tarea 1. \href{https://github.com/HazelS002/EDP-II/tree/main}
        {Consultar reposritorio aqui}\eej

        \item Con el algoritmo creado resolver los siguientes problemas, compara
        con las soluciones analiticas presentadas por Chen en 2004:
        \begin{enumerate}[(a)]
            \item $y'=x^2+y^2$, $y(0)=0$
            \item $u_t=x^2-(u_x)^2/4$, $u(x,0)=0$
            \item $u_t+u^2u_x=0$, $u(x,0)=3x$
            \item \begin{align*}
                \frac{d^2}{dt^2}u+v=0,&\qquad \frac{d^2}{dt^2}v+u=0,\\
                u(0)=0,&\quad u'(0)=1,\\
                v(0)=0,&\quad v'(0)=-1.
            \end{align*}
            \item \begin{align*}
                \frac{\partial^2}{\partial t^2}u=
                \frac{\partial^2}{\partial x^2}u,&\quad 0<x<\pi,~t>0,\\
                u(0,t)=u(\pi, t)=0,&\quad t>0,\\
                u(x,0)=\sin^3(x),&\quad 0<x<\pi,\\
                u_t(x,0) = \sin(2x),&\quad 0<x<\pi.
            \end{align*}
        \end{enumerate}

        \item Resolver los problemas de Black-Scholes:
        \begin{enumerate}[(a)]
            \item \[
                \begin{cases}
                    rC(t, x) = C_t(t, x) + \frac{1}{2}\sigma^2x^2C_{xx}(t, x)
                    + rxC_x(t, x),\qquad& x > 0,~t\in [0, T],\\
                    C(T, x) = \max(x-K, 0),\qquad&\\
                    C(t, x) = x-Ke^{-r(T-t)},\qquad&\text{cuando }x\to\infty,\\
                    C(t, 0) = 0,\qquad&\forall t > 0.
                \end{cases}
            \]

            \sol Se introducen los siguientes cambios de variable 
            \[ \tau = \frac{1}{2}\sigma^2 (T-t),\quad y = \log\pp{\frac{x}{K}},
            \quad C(t,x) = K\,V(\tau,y). \]
            Sustituyendo en la ecuación original tenemos
            \[ -V_\tau + V_{yy} + \pp{\frac{2r}{\sigma^2}-1}V_y =
            \frac{2r}{\sigma^2} V. \]
            Definimos
            \[ a = \frac{1}{2}\pp{\frac{2r}{\sigma^2}-1} = \frac{r}{\sigma^2}
            -\frac{1}{2},\qquad b = (1+a)^2. \]
            Mediante la transformación 
            \[ U(\tau,y) = e^{a y + b\tau}\, V(\tau,y), \]
            la ecuación se reduce a la ecuación del calor
            \[ U_\tau = U_{yy}, \qquad y>0,~\tau>0. \]
            Y las condiciones inicial y de frontera se transforman en
            \[ \begin{cases}
                    U(0,y) = \max\pp{e^{\frac{1}{2}(a+1)y}
                    -e^{\frac{1}{2}(a-1)y},~0},\\
                    U(\tau,0) = 0,\\
                    U(\tau,L) = e^{\frac{1}{2}(a+1)L + \frac{1}{4}(a+1)^2\tau}
                    -e^{\frac{1}{2}(a-1)L + \frac{1}{4}(a-1)^2\tau},
                \end{cases} \]
            donde $L$ es un valor suficientemente grande para aproximar la
            condición en el infinito.

            Dado que el MDA se aplica directamente a problemas con condiciones
            de Dirichlet homogéneas, se descompone
            \[ U(\tau,y) = u(\tau,y) + w(\tau,y), \]
            con $w(\tau,y)$ elegida de modo que $u$ satisfaga condiciones nulas
            en los bordes. Tomamos
            \[ w(\tau,y) = U(0,y) + \pp{U(0,y)-U(\tau,L)}\frac{y}{L}. \]
            Se tiene que
            \[ w(\tau,0)=U(0,0)=0,\qquad w(\tau,L)=U(\tau,L). \]
            Se obtiene el problema para $u$
            \[ \begin{cases}
                u_\tau = u_{yy} - w_\tau, & y\in(0,L),~\tau>0,\\
                u(0,y) = U(0,y)-w(0,y)=0,&\\
                u(\tau,0)= u(\tau,L)=0.&
            \end{cases} \]
            Luego, definimos el operador integral
            \[ L^{-1} = \int_0^\tau (\cdot)\,d\tau'. \]
            Integrando la ecuación
            \[ u(\tau,y) = u(0,y) - L^{-1}w_\tau + \int_0^\tau u_{yy}(s,y)ds =
            -L^{-1}w_\tau + \int_0^\tau u_{yy}(s,y)ds, \]
            ya que $u(0,y)=0$.

            Suponemos que
            \[ u(\tau,y) = \sum_{i=0}^\infty u_i(\tau,y). \]
            Sustituyendo y usando la linealidad del operador, se obtiene
            la recurrencia de Adomian para un problema lineal
            \begin{align}
                u_0(\tau,y) &= -L^{-1}w_\tau(\tau,y),\\
                u_{n+1}(\tau,y) &= \int_0^\tau \frac{\partial^2}{\partial y^2}
                u_n(s,y)ds, \qquad n\ge 0.
            \end{align}
            Los términos se calculan explícitamente:
            \begin{itemize}
                \item $u_0 = w(0,y)-w(\tau,y) = U(0,y)-w(\tau,y)$.
                \item $u_1 = \int_0^\tau (u_0)_{yy}\,ds$.
                \item Y en general, para $n\ge 1$,
                \[ u_n(\tau,y) = \frac{\tau^n}{n!}
                \frac{\partial^{2n} u_0}{\partial y^{2n}}(0,y), \]
            \end{itemize}
            La aproximación de orden $k+1$ es
            \[ \psi_k(\tau,y) = \sum_{i=0}^{k} u_i(\tau,y) \approx u(\tau,y). \]
            Una vez obtenida $u(\tau,y)$ (aproximadamente), se reconstruye:
            \[ U(\tau,y) = u(\tau,y) + w(\tau,y),\qquad V(\tau,y)
            = e^{-a y - b\tau} U(\tau,y), \]
            y finalmente el precio de la opción de compra
            \[ C(t,x) = K e^{-\frac{1}{2}(a+1)y - \frac{1}{4}(a-1)^2\tau}
            \;U(\tau,y), \]
            donde  $a = \dfrac{r}{\sigma^2}-\dfrac{1}{2}$.


            \item Consideremos el problema:
            \[
                \begin{cases}
                    U_t(\tau, y) = U_{xx}(\tau, y),\quad& y > 0,
                    \tau\in\cc{0, \tfrac{\sigma^2T}{2}},\\
                    U(0, y) = \max\pp{e^{\tfrac{1}{2}(\gamma+1)y}
                    -e^{\tfrac{1}{2}(\gamma-1)y},0},\\
                    U(\tau, L) = e^{\tfrac{1}{2}(\gamma+1)L+\tfrac{1}{4}
                    (\gamma+1)^2\tau}-e^{\tfrac{1}{2}(\gamma-1)L
                    +\tfrac{1}{4}(\gamma-1)^2\tau},\\
                    U(\tau, 0) = 0,
                \end{cases}
            \]
            con
            \[ \tau=\frac{1}{2}\sigma^2(T-t),\qquad y=\log\pp{\frac{x}{K}},
            \qquad \gamma=\frac{2r}{\sigma^2}. \]
            Recordemos que al Usar MDA no se usan las condiciones en la frontera
            como en a), por lo cual se hace una modificación usando
            \begin{align*}
                U(\tau,y) &= u(\tau,y)+w(\tau,y),\\
                w(\tau,y) &= U(0,y)+\pp{U(0,y)-U(\tau,L)}
                \pp{\frac{y-y_0}{L-y_0}}
            \end{align*}
            el problema se convierte en uno dónde las condiciones de frontera
            son nulas, en este caso, se puede usar el ADM tradicional que usa
            sólo la condición inicial:
            \[
                \begin{cases}
                    u_t(\tau,y)=u_{xx}(\tau, y)-w_t(\tau,y),\quad y>0,
                    ~\tau\in\cc{0,\tfrac{\sigma^2T}{2}},\\
                    u(0,y) = u_0(y)-w(0,y),\\
                    u(\tau,L)=0,\\
                    u(\tau,0)=0,\quad \forall \tau >0,
                \end{cases}
            \]
            con
            \[ u_0(y) = \max\pp{e^{\tfrac{1}{2}(\gamma+1)y}
            -e^{\tfrac{1}{2}(\gamma-1)y}, 0}. \]
            Con el ADM obtenemos:
            \[ u(t)\approx\psi_k = \sum_{i=0}^{k-1}u_i(t). \]
            La solución es
            \[ C(t, x)=Ke^{-\tfrac{1}{2}(\gamma+1)x
            -\tfrac{1}{4}(\gamma-1)^{2}\tau}\pp{\psi_k+w(\tau,y)}. \]
            Usa $r=0,05$, $\sigma=0.317$, $K=20$ y $T=0,25$ (3 meses), compara
            los resultados.

        \end{enumerate}
    \end{enumerate}
